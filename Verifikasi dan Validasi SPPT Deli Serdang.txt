SELECT A.THN_KEGIATAN,C.NM_KELURAHAN, D.NM_KECAMATAN, A.KD_KELURAHAN, A.KD_PROPINSI||'.'||A.KD_DATI2||'.'||A.KD_KELURAHAN||'.'||A.KD_BLOK||'-'||A.NO_URUT||'.'||A.KD_JNS_OP AS NOP, B.NM_WP_SPPT ||' / '||A.JALAN_OP||'-'||A.BLOK_KAV_NO_OP AS WAJIB_PAJAK_LETAK_OBJEK_PAJAK, 'K'||A.KATEGORI_OP AS KAT_OBJEK, 'Rp. '||TRIM(TO_CHAR(SUM(B.PBB_YG_HRS_DIBAYAR_SPPT),'99G999G999G999', 'NLS_NUMERIC_CHARACTERS=".,"')) AS TOTAL_TUNGGAKAN
  FROM DAFNOM_OP A
    INNER JOIN SPPT B ON
                        A.KD_PROPINSI = B.KD_PROPINSI AND A.KD_DATI2 = B.KD_DATI2 AND A.KD_KECAMATAN = B.KD_KECAMATAN AND A.KD_KELURAHAN = B.KD_KELURAHAN AND A.KD_BLOK = B.KD_BLOK AND A.NO_URUT = B.NO_URUT AND A.KD_JNS_OP = B.KD_JNS_OP
    INNER JOIN IPROTAXREF.REF_KELURAHAN C ON
      A.KD_KELURAHAN = C.KD_KELURAHAN AND A.KD_KECAMATAN = C.KD_KECAMATAN
    INNER JOIN IPROTAXREF.REF_KECAMATAN D ON
      A.KD_KECAMATAN = D.KD_KECAMATAN
GROUP BY A.KD_PROPINSI, A.KD_DATI2, A.KD_KECAMATAN, A.KD_KELURAHAN, A.KD_BLOK, A.NO_URUT, A.KD_JNS_OP, A.JALAN_OP, A.BLOK_KAV_NO_OP, A.KATEGORI_OP, A.THN_KEGIATAN,
  B.KD_PROPINSI, B.KD_DATI2, B.KD_KECAMATAN, B.KD_KELURAHAN, B.KD_BLOK, B.NO_URUT, B.KD_JNS_OP, B.NM_WP_SPPT,
  C.KD_KELURAHAN,C.NM_KELURAHAN, D.KD_KECAMATAN, D.NM_KECAMATAN
ORDER BY A.KD_PROPINSI, A.KD_DATI2, A.KD_KECAMATAN, A.KD_KELURAHAN, A.KD_BLOK, A.NO_URUT, A.KD_JNS_OP, A.THN_KEGIATAN;

SELECT 'K'||A.KATEGORI_OP AS KAT_OBJEK, 'Rp. '||TRIM(TO_CHAR(SUM(B.PBB_YG_HRS_DIBAYAR_SPPT),'99G999G999G999', 'NLS_NUMERIC_CHARACTERS=".,"')) AS TOTAL_TUNGGAKAN
  FROM DAFNOM_OP A INNER JOIN SPPT B ON A.KD_PROPINSI = B.KD_PROPINSI AND A.KD_DATI2 = B.KD_DATI2 AND A.KD_KECAMATAN = B.KD_KECAMATAN AND A.KD_KELURAHAN = B.KD_KELURAHAN AND A.KD_BLOK = B.KD_BLOK AND A.NO_URUT = B.NO_URUT AND A.KD_JNS_OP = B.KD_JNS_OP
GROUP BY A.KATEGORI_OP ORDER BY A.KATEGORI_OP;

SELECT 'Rp. '||TRIM(TO_CHAR(SUM(B.PBB_YG_HRS_DIBAYAR_SPPT),'99G999G999G999', 'NLS_NUMERIC_CHARACTERS=".,"')) AS TOTAL_TUNGGAKAN
  FROM DAFNOM_OP A INNER JOIN SPPT B ON A.KD_PROPINSI = B.KD_PROPINSI AND A.KD_DATI2 = B.KD_DATI2 AND A.KD_KECAMATAN = B.KD_KECAMATAN AND A.KD_KELURAHAN = B.KD_KELURAHAN AND A.KD_BLOK = B.KD_BLOK AND A.NO_URUT = B.NO_URUT AND A.KD_JNS_OP = B.KD_JNS_OP
;
